# Plan: 自定义脚手架生成器 — Admin & API

## Context

zen_platform 需要对标 Rails 官方脚手架功能，创建自定义的 Admin 脚手架和 API 脚手架生成器。项目使用 Inertia.js + React (Ant Design Pro) 作为前端，API 层使用自定义的 ApiController 基类。

---

## 当前架构分析

### 控制器继承链
```
AdminController (session 认证)
  └─ Admin::* 控制器

ApiController (混合认证 + 统一响应)
  └─ Api::V1::* 控制器
```

### 前端架构
- **Admin 后台**：Inertia.js + React + Ant Design Pro
- **视图位置**：`app/frontend/pages/admin/`
- **布局组件**：`app/frontend/layouts/AdminLayout.tsx`

### 2. Admin 控制器模板
```ruby
# frozen_string_literal: true

module Admin
  class <%= class_name.pluralize %>Controller < AdminController
    before_action :require_login
    before_action :set_<%= singular_name %>, only: [:show, :edit, :update, :destroy]

    def index
      @<%= plural_name %> = <%= class_name %>.all
      render inertia: "admin/<%= plural_name.camelize %>/Index",
        props: { <%= plural_name %>: @<%= plural_name %>.as_json }
    end

    def show
      render inertia: "admin/<%= plural_name.camelize %>/Show",
        props: { <%= singular_name %>: @<%= singular_name %>.as_json }
    end

    def new
      @<%= singular_name %> = <%= class_name %>.new
      render inertia: "admin/<%= plural_name.camelize %>/New"
    end

    def create
      @<%= singular_name %> = <%= class_name %>.new(<%= singular_name %>_params)
      if @<%= singular_name %>.save
        redirect_to admin_<%= plural_name %>_path(@<%= singular_name %>), notice: "创建成功"
      else
        render inertia: "admin/<%= plural_name.camelize %>/New",
          props: { errors: @<%= singular_name %>.errors.full_messages }
      end
    end

    def edit
      render inertia: "admin/<%= plural_name.camelize %>/Edit",
        props: { <%= singular_name %>: @<%= singular_name %>.as_json }
    end

    def update
      if @<%= singular_name %>.update(<%= singular_name %>_params)
        redirect_to admin_<%= plural_name %>_path(@<%= singular_name %>), notice: "更新成功"
      else
        render inertia: "admin/<%= plural_name.camelize %>/Edit",
          props: { <%= singular_name %>: @<%= singular_name %>.as_json, errors: @<%= singular_name %>.errors.full_messages }
      end
    end

    def destroy
      @<%= singular_name %>.destroy
      redirect_to admin_<%= plural_name %>_path, notice: "删除成功"
    end

    private

    def set_<%= singular_name %>
      @<%= singular_name %> = <%= class_name %>.find(params[:id])
    end

    def <%= singular_name %>_params
      params.require(:<%= singular_name %>).permit(<%= attributes.map { |a| ":#{a.name}" }.join(", ") %>)
    end
  end
end
```

### 3. API 控制器模板
```ruby
# frozen_string_literal: true

module Api
  module V1
    class <%= class_name.pluralize %>Controller < ApiController
      include Api::JwtAuthenticatable
      include Api::SwaggerDocControllable

      swagger_tag "<%= class_name %>管理"

      before_action { authenticate_with :jwt }

      def index
        @<%= plural_name %> = <%= class_name %>.all
        render_success @<%= plural_name %>.as_json
      end

      def show
        @<%= singular_name %> = <%= class_name %>.find_by(id: params[:id])
        if @<%= singular_name %>
          render_success @<%= singular_name %>.as_json
        else
          render_error(message: "记录不存在", status: :not_found)
        end
      end

      def create
        @<%= singular_name %> = <%= class_name %>.new(<%= singular_name %>_params)
        if @<%= singular_name %>.save
          render_success @<%= singular_name %>.as_json
        else
          render_error(message: @<%= singular_name %>.errors.full_messages.join(", "))
        end
      end

      def update
        @<%= singular_name %> = <%= class_name %>.find_by(id: params[:id])
        if @<%= singular_name %>.nil?
          render_error(message: "记录不存在", status: :not_found)
        elsif @<%= singular_name %>.update(<%= singular_name %>_params)
          render_success @<%= singular_name %>.as_json
        else
          render_error(message: @<%= singular_name %>.errors.full_messages.join(", "))
        end
      end

      def destroy
        @<%= singular_name %> = <%= class_name %>.find_by(id: params[:id])
        if @<%= singular_name %>
          @<%= singular_name %>.destroy
          render_success
        else
          render_error(message: "记录不存在", status: :not_found)
        end
      end

      private

      def <%= singular_name %>_params
        params.require(:<%= singular_name %>).permit(<%= attributes.map { |a| ":#{a.name}" }.join(", ") %>)
      end
    end
  end
end
```

### 4. React 页面模板（Index 示例）
```tsx
import { useState, useEffect } from 'react'
import { router } from '@inertiajs/react'
import { PageContainer, ProTable } from '@ant-design/pro-components'
import { Button, Space, Popconfirm, message } from 'antd'
import { PlusOutlined, EditOutlined, DeleteOutlined, EyeOutlined } from '@ant-design/icons'
import AdminLayout from '../../../layouts/AdminLayout'
import type { ReactNode } from 'react'

interface <%= class_name %> {
<%= attributes.map { |a| "  #{a.name}: #{a.type == 'integer' ? 'number' : 'string'}" }.join("\n") %>
}

<%= plural_name.camelize %>.layout = (page: ReactNode) => <AdminLayout>{page}</AdminLayout>

export default function <%= plural_name.camelize %>({ <%= plural_name %> }: { <%= plural_name %>: <%= class_name %>[] }) {
  const columns = [
<%= attributes.map { |a| "    { title: '#{a.name}', dataIndex: '#{a.name}', key: '#{a.name}' }" }.join(",\n") %>,
    {
      title: '操作',
      key: 'action',
      render: (_: any, record: <%= class_name %>) => (
        <Space>
          <Button
            type="link"
            icon={<EyeOutlined />}
            onClick={() => router.visit(`/admin/<%= plural_name %>/${record.id}`)}
          />
          <Button
            type="link"
            icon={<EditOutlined />}
            onClick={() => router.visit(`/admin/<%= plural_name %>/${record.id}/edit`)}
          />
          <Popconfirm
            title="确定删除？"
            onConfirm={() => {
              router.delete(`/admin/<%= plural_name %>/${record.id}`, {
                onSuccess: () => message.success('删除成功'),
              })
            }}
          >
            <Button type="link" danger icon={<DeleteOutlined />} />
          </Popconfirm>
        </Space>
      ),
    },
  ]

  return (
    <PageContainer title="<%= class_name %>管理">
      <ProTable
        columns={columns}
        dataSource={<%= plural_name %>}
        rowKey="id"
        toolBarRender={() => [
          <Button
            key="add"
            type="primary"
            icon={<PlusOutlined />}
            onClick={() => router.visit('/admin/<%= plural_name %>/new')}
          >
            新建
          </Button>,
        ]}
      />
    </PageContainer>
  )
}
```
