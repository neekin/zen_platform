一行 DSL，生成整个后台

在 Model 中声明一次，前端自动渲染。无需手写 CRUD。

app/models/article.rb
class Article < ApplicationRecord
  include Zen::ModelDsl

  field :title, :string, required: true
  field :body, :rich_text
  field :status, :enum, values: %w[draft published]

  belongs_to :category

  display do
    list do
      column :title, link: true
      column :status, badge: true
    end
    form do
      field :title
      field :body, as: :rich_text
    end
  end
end
app/controllers/admin/articles_controller.rb
# 自动生成，无需手写
class Admin::ArticlesController < AdminController
  include Pagy::Method

  def index
    q = Article.ransack(search_params)
    @pagy, articles = pagy(:offset, q.result, **pagy_params)
    render inertia: "admin/articles/Index",
      props: zen_props(Article, articles: articles)
  end
end
